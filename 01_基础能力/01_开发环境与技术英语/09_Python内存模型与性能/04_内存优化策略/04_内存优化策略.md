## 正文讲解

### 1. 为什么需要优化内存？

在处理小数据集时，Python 的内存使用通常不是问题。但在 AI 和数据科学领域，你经常会遇到这些场景：

- 加载百万级的训练样本到内存中
- 处理 GB 级别的日志文件
- 同时运行多个实验，内存成为瓶颈
- 在内存有限的服务器上部署模型

Python 对象的 **内存开销（Overhead）** 比你想象的要大。一个简单的整数 `42` 在 C 语言中只需 4 字节，但在 Python 中需要 28 字节！这是因为 Python 的每个对象都携带了类型信息、引用计数器等元数据。

```python
# 文件：code/memory_overhead.py
import sys

# Python 对象的内存开销远超你的想象
print("=== 单个对象的内存开销 ===")
print(f"  int(0):     {sys.getsizeof(0)} bytes")       # 28
print(f"  int(42):    {sys.getsizeof(42)} bytes")      # 28
print(f"  float(3.14):{sys.getsizeof(3.14)} bytes")   # 24
print(f"  str(''):    {sys.getsizeof('')} bytes")      # 49
print(f"  str('hi'):  {sys.getsizeof('hi')} bytes")   # 51
print(f"  list([]):   {sys.getsizeof([])} bytes")     # 56
print(f"  dict({{}}):   {sys.getsizeof({})} bytes")    # 64
print(f"  tuple(()):  {sys.getsizeof(())} bytes")     # 40

# 100 万个整数的列表
big_list = list(range(1_000_000))
list_size = sys.getsizeof(big_list)  # 列表壳的大小
# 加上每个元素的大小（整数对象）
total_size = list_size + sum(sys.getsizeof(x) for x in big_list[:100]) / 100 * len(big_list)
print(f"\n  100 万整数列表:")
print(f"    列表壳: {list_size / 1024 / 1024:.1f} MB")
print(f"    估算总内存: {total_size / 1024 / 1024:.1f} MB")
```

**运行说明**：
- 环境要求：Python 3.10+
- 运行命令：`python code/memory_overhead.py`

接下来，我们逐一学习几种有效的内存优化策略。

### 2. __slots__——减少实例内存

默认情况下，Python 的每个实例都有一个 `__dict__` 字典来存储属性。这个字典很灵活（可以动态添加属性），但占用了大量内存。使用 `__slots__` 可以告诉 Python："这个类的实例只有这些固定属性"，从而用更紧凑的结构替代字典。

```python
# 文件：code/slots_demo.py
import sys

class PointDict:
    """普通类——使用 __dict__ 存储属性"""
    def __init__(self, x, y, z):
        self.x = x
        self.y = y
        self.z = z

class PointSlots:
    """使用 __slots__ 的类——禁用 __dict__"""
    __slots__ = ("x", "y", "z")

    def __init__(self, x, y, z):
        self.x = x
        self.y = y
        self.z = z

# 比较单个实例的大小
p_dict = PointDict(1.0, 2.0, 3.0)
p_slots = PointSlots(1.0, 2.0, 3.0)

size_dict = sys.getsizeof(p_dict) + sys.getsizeof(p_dict.__dict__)
size_slots = sys.getsizeof(p_slots)

print(f"普通类实例: {size_dict} bytes（对象 {sys.getsizeof(p_dict)} + __dict__ {sys.getsizeof(p_dict.__dict__)}）")
print(f"__slots__ 实例: {size_slots} bytes")
print(f"节省: {(1 - size_slots / size_dict) * 100:.0f}%")

# 大规模对比
n = 1_000_000
print(f"\n创建 {n:,} 个实例...")

import tracemalloc
tracemalloc.start()

points_dict = [PointDict(i, i+1, i+2) for i in range(n)]
snapshot_dict = tracemalloc.take_snapshot()
mem_dict = sum(stat.size for stat in snapshot_dict.statistics("filename"))

# 清理并重新测量
del points_dict
tracemalloc.stop()
tracemalloc.start()

points_slots = [PointSlots(i, i+1, i+2) for i in range(n)]
snapshot_slots = tracemalloc.take_snapshot()
mem_slots = sum(stat.size for stat in snapshot_slots.statistics("filename"))

print(f"普通类: {mem_dict / 1024 / 1024:.1f} MB")
print(f"__slots__: {mem_slots / 1024 / 1024:.1f} MB")
print(f"节省: {(1 - mem_slots / mem_dict) * 100:.0f}%")

tracemalloc.stop()

# __slots__ 的限制
try:
    p_slots.w = 4.0  # 无法动态添加属性
except AttributeError as e:
    print(f"\n__slots__ 限制: {e}")
```

**预期输出**（具体数字因平台而异）：
```
普通类实例: 200 bytes（对象 48 + __dict__ 152）
__slots__ 实例: 64 bytes
节省: 68%
```

> 💡 **使用建议**：当你需要创建大量（万级以上）相同结构的对象时，使用 `__slots__` 可以显著降低内存。但如果只创建少量对象，额外的约束（不能动态添加属性）可能不值得。

### 3. 生成器——惰性计算避免内存爆炸

**[生成器（Generator）](../../01_编程语言基础/04_列表推导与生成器/)** 是 Python 中最强大的内存优化工具之一。它不像列表那样一次性把所有数据加载到内存中，而是"按需生成"——每次只产出一个元素。

```python
# 文件：code/generator_memory.py
import sys

# === 列表：一次性加载全部数据 ===
def get_squares_list(n):
    """返回列表——全部数据在内存中"""
    return [x ** 2 for x in range(n)]

# === 生成器：按需产出 ===
def get_squares_gen(n):
    """返回生成器——一次只有一个元素在内存中"""
    for x in range(n):
        yield x ** 2

n = 10_000_000

# 比较内存占用
squares_list = get_squares_list(n)
squares_gen = get_squares_gen(n)

print(f"列表占用: {sys.getsizeof(squares_list) / 1024 / 1024:.1f} MB")
print(f"生成器占用: {sys.getsizeof(squares_gen)} bytes")

# 两者功能相同——都可以遍历
print(f"\n列表前5个: {squares_list[:5]}")
print(f"生成器前5个: {[next(squares_gen) for _ in range(5)]}")

# 生成器表达式（更简洁的写法）
gen_expr = (x ** 2 for x in range(n))  # 圆括号而非方括号
print(f"\n生成器表达式占用: {sys.getsizeof(gen_expr)} bytes")
```

**预期输出**：
```
列表占用: 76.3 MB
生成器占用: 200 bytes

列表前5个: [0, 1, 4, 9, 16]
生成器前5个: [0, 1, 4, 9, 16]

生成器表达式占用: 200 bytes
```

生成器特别适合以下场景：

- **逐行读取大文件**：不需要把整个文件加载到内存
- **数据管道**：一系列转换操作可以用生成器串联
- **无限序列**：如斐波那契数列、日志流等

```python
# 文件：code/generator_pipeline.py
import os

def read_lines(filename):
    """逐行读取文件（生成器）"""
    with open(filename, "r") as f:
        for line in f:
            yield line.strip()

def filter_nonempty(lines):
    """过滤空行（生成器）"""
    for line in lines:
        if line:
            yield line

def to_upper(lines):
    """转大写（生成器）"""
    for line in lines:
        yield line.upper()

# 创建一个临时测试文件
test_file = "/tmp/test_generator.txt"
with open(test_file, "w") as f:
    for i in range(100):
        f.write(f"line {i}\n")
        if i % 10 == 0:
            f.write("\n")  # 插入空行

# 生成器管道——内存使用恒定，不随文件大小增长
pipeline = to_upper(filter_nonempty(read_lines(test_file)))

# 只取前 5 个结果
for i, line in enumerate(pipeline):
    if i >= 5:
        break
    print(f"  {line}")

# 清理
os.remove(test_file)
```

### 4. 高效数据结构

不同的数据结构在内存效率上差异巨大。选择合适的数据结构是内存优化的基本功：

```python
# 文件：code/efficient_structures.py
import sys
from array import array
from collections import namedtuple

n = 100_000

# === 整数列表 vs array ===
int_list = list(range(n))
int_array = array("l", range(n))  # 'l' = signed long

print("=== 整数存储 ===")
print(f"  list: {sys.getsizeof(int_list) / 1024:.1f} KB（不含元素对象）")
print(f"  array: {sys.getsizeof(int_array) / 1024:.1f} KB（紧凑存储）")

# === 浮点数列表 vs array ===
float_list = [float(i) for i in range(n)]
float_array = array("d", (float(i) for i in range(n)))  # 'd' = double

print("\n=== 浮点数存储 ===")
print(f"  list: {sys.getsizeof(float_list) / 1024:.1f} KB（不含元素对象）")
print(f"  array: {sys.getsizeof(float_array) / 1024:.1f} KB（紧凑存储）")

# === 字典 vs namedtuple ===
Record = namedtuple("Record", ["name", "age", "score"])

dict_records = [{"name": f"user_{i}", "age": 20 + i % 50, "score": i * 0.1} for i in range(n)]
tuple_records = [Record(f"user_{i}", 20 + i % 50, i * 0.1) for i in range(n)]

dict_size = sys.getsizeof(dict_records) + sum(sys.getsizeof(d) for d in dict_records[:100]) / 100 * n
tuple_size = sys.getsizeof(tuple_records) + sum(sys.getsizeof(t) for t in tuple_records[:100]) / 100 * n

print(f"\n=== 记录存储（{n:,} 条） ===")
print(f"  dict 列表: ~{dict_size / 1024 / 1024:.1f} MB")
print(f"  namedtuple 列表: ~{tuple_size / 1024 / 1024:.1f} MB")

# === 如果有 NumPy ===
try:
    import numpy as np
    np_array = np.arange(n, dtype=np.int64)
    print(f"\n=== NumPy 对比 ===")
    print(f"  Python list: {sys.getsizeof(int_list) / 1024:.1f} KB")
    print(f"  array.array: {sys.getsizeof(int_array) / 1024:.1f} KB")
    print(f"  numpy array: {np_array.nbytes / 1024:.1f} KB")
except ImportError:
    print("\n（NumPy 未安装，跳过 NumPy 对比）")
```

**运行说明**：
- 环境要求：Python 3.10+，numpy（可选）
- 运行命令：`python code/efficient_structures.py`

关键结论：

| 场景 | 推荐结构 | 原因 |
| ---- | -------- | ---- |
| 大量同类型数值 | `array.array` 或 NumPy | 紧凑存储，无 Python 对象开销 |
| 结构化记录 | `namedtuple` 或 `dataclass(slots=True)` | 比字典省内存 |
| 需要快速查找 | `set` 或 `frozenset` | $O(1)$ 查找，比列表省时 |
| 键值映射（大量小字典） | 合并为 DataFrame 或 array | 减少字典开销 |

### 5. 内存映射——处理超大文件

当文件大到无法完全加载到内存时，**内存映射（Memory-Mapped File, mmap）** 是一个强大的选择。它让你可以像访问数组一样访问文件内容，而操作系统会自动管理哪些部分加载到物理内存中：

```python
# 文件：code/mmap_demo.py
import mmap
import os

# 创建一个测试文件
test_file = "/tmp/test_mmap.bin"
with open(test_file, "wb") as f:
    # 写入 10 MB 的数据
    for i in range(1_000_000):
        f.write(f"Line {i:08d}: Some data here\n".encode("utf-8"))

file_size = os.path.getsize(test_file)
print(f"文件大小: {file_size / 1024 / 1024:.1f} MB")

# 使用 mmap 访问文件——不需要全部加载到内存
with open(test_file, "r+b") as f:
    # 将文件映射到内存
    with mmap.mmap(f.fileno(), 0) as mm:
        # 读取前 100 个字节
        print(f"前 100 字节: {mm[:100].decode('utf-8')}")

        # 搜索特定内容
        pos = mm.find(b"Line 00099999")
        if pos != -1:
            mm.seek(pos)
            line = mm.readline()
            print(f"找到: {line.decode('utf-8').strip()}")

        # 获取文件大小
        print(f"映射大小: {mm.size()} bytes")

# 清理
os.remove(test_file)
print("\n测试文件已清理")
```

### 6. 综合优化策略速查表

| 策略 | 适用场景 | 节省效果 | 实现难度 |
| ---- | -------- | -------- | -------- |
| `__slots__` | 大量相同结构的对象 | 40-60% | ⭐ |
| 生成器替代列表 | 流式处理大数据 | 90%+ | ⭐ |
| `array` 替代 `list` | 大量同类型数值 | 60-80% | ⭐ |
| NumPy 数组 | 数值计算和矩阵操作 | 80-90% | ⭐⭐ |
| `namedtuple` 替代字典 | 结构化记录 | 30-50% | ⭐ |
| `mmap` 内存映射 | 超大文件处理 | 取决于访问模式 | ⭐⭐ |
| 及时 `del` 和 GC | 临时大对象 | 视情况而定 | ⭐ |
| 数据分批处理 | 数据集超过内存 | 不受内存限制 | ⭐⭐ |

## 典型误区

| 误区 | 正确理解 |
| ---- | -------- |
| `__slots__` 是为了加速 | `__slots__` 主要是为了节省内存，属性访问速度提升通常很小 |
| 生成器总是比列表好 | 如果需要多次遍历或随机访问，列表更合适。生成器只能遍历一次 |
| `sys.getsizeof()` 显示的是全部内存 | `getsizeof` 只测量对象自身的大小，不包括它引用的其他对象 |
| 内存优化应该无处不在 | 只在确实遇到内存问题时才优化。过度优化会降低代码可读性 |

### 练习 2：生成器管道处理日志（难度：⭐⭐）

使用生成器实现一个日志处理管道：
1. 逐行读取日志文件
2. 过滤出包含 "ERROR" 的行
3. 提取时间戳和错误信息
4. 只输出前 10 条

要求：整个过程的内存占用应恒定（不随文件大小增长）。

<details>
<summary>💡 提示</summary>

定义三个生成器函数，每个负责一个处理步骤，然后用管道方式串联。使用 `itertools.islice` 或手动计数来限制输出。

</details>

<details>
<summary>✅ 参考答案</summary>

```python
import itertools
import os

# 创建模拟日志文件
log_file = "/tmp/test.log"
with open(log_file, "w") as f:
    for i in range(10000):
        level = "ERROR" if i % 100 == 0 else "INFO"
        f.write(f"2025-04-15 10:{i//60:02d}:{i%60:02d} [{level}] Message {i}\n")

def read_lines(path):
    with open(path) as f:
        for line in f:
            yield line.strip()

def filter_errors(lines):
    for line in lines:
        if "ERROR" in line:
            yield line

def extract_info(lines):
    for line in lines:
        parts = line.split(" ", 2)
        timestamp = f"{parts[0]} {parts[1]}"
        message = parts[2] if len(parts) > 2 else ""
        yield {"time": timestamp, "message": message}

# 管道处理
pipeline = extract_info(filter_errors(read_lines(log_file)))
for entry in itertools.islice(pipeline, 10):
    print(f"  {entry['time']} | {entry['message']}")

os.remove(log_file)
```

</details>

### 练习 3：内存占用对比报告（难度：⭐⭐）

编写一个函数，接受一个整数 `n`，分别用以下方式存储 `n` 个整数，并打印每种方式的内存占用：
1. Python 列表
2. `array.array`
3. NumPy 数组（如果可用）
4. 生成器（只打印生成器对象本身的大小）

<details>
<summary>💡 提示</summary>

使用 `sys.getsizeof()` 测量每种结构。对于包含元素的结构，还需要估算元素的内存。

</details>

<details>
<summary>✅ 参考答案</summary>

```python
import sys
from array import array

def memory_report(n):
    print(f"存储 {n:,} 个整数的内存对比:")

    # 列表
    lst = list(range(n))
    print(f"  list:      {sys.getsizeof(lst) / 1024:.1f} KB（仅壳）")

    # array
    arr = array("l", range(n))
    print(f"  array:     {sys.getsizeof(arr) / 1024:.1f} KB（含数据）")

    # 生成器
    gen = (x for x in range(n))
    print(f"  generator: {sys.getsizeof(gen)} bytes（惰性）")

    # NumPy
    try:
        import numpy as np
        npa = np.arange(n, dtype=np.int64)
        print(f"  numpy:     {npa.nbytes / 1024:.1f} KB（含数据）")
    except ImportError:
        print("  numpy:     未安装")

memory_report(1_000_000)
```

</details>
